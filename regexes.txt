тип переменной
:'<,'>s/\([a-zA-Z_0-9]\+\) =/auto \1 =/g

функция
:'<,'>s/def \(.*\)(\(.*\)):/Matrix2d \1(\2) {/g

циклы
:%s/for \([a-z]\+\) in range(\([0-9]\+\),\s*\([0-9]\+\)):/for (int \1 = \2; \1 < \2; ++\1)/g
:%s/while \(.*\):/while (\1)/g

условия
:%s/if \(.*\):/if (\1)/g

вывод
:%s/print(\(.*\))/cout << \1 << endl;/g

комменты
:%s/.*In\[[0-9]\+\].*//g
:'<,'>s/#/\/\//g
